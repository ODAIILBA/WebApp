### 4. **Generation Workflow**

#### **Step-by-Step Process**

```
1. TRIGGER
   - WP-Cron scheduled event
   - Manual trigger from admin
   - REST API endpoint call

2. SEARCH PHASE
   - Query web search API with configured topics
   - Example: "WordPress plugins 2026", "AI trends", etc.
   - Fetch top 5-10 results
   - Extract titles, snippets, URLs
   - Filter and rank by relevance

3. CONTENT GENERATION PHASE
   - Prepare AI prompt with search results
   - Call OpenAI API (GPT-4)
   - Request structured JSON output:
     {
       "title": "Post title",
       "content": "<p>HTML content...</p>",
       "excerpt": "Short summary",
       "meta_description": "SEO description",
       "keywords": ["keyword1", "keyword2"],
       "tags": ["tag1", "tag2"],
       "category": "category-slug"
     }
   - Parse and validate response

4. IMAGE GENERATION (Optional)
   - Generate featured image with DALL-E 3
   - Or use Pexels/Unsplash API for stock photos
   - Upload to WordPress media library
   - Set as post featured image

5. POST CREATION
   - Create post with wp_insert_post()
   - Set title, content, excerpt
   - Assign category, tags
   - Set featured image
   - Add AI meta data
   - Set post status (publish or draft)
   - Set post author

6. LOGGING
   - Save generation log to custom table
   - Record success/failure
   - Store API response
   - Track tokens and costs

7. NOTIFICATIONS (Optional)
   - Email admin on success/failure
   - Slack webhook notification
   - WordPress admin notice
```

### **Core Class Example: AI Generator**

```php
<?php
/**
 * AI Content Generation Class
 *
 * @package AI_Auto_Blogger_Pro
 */

class AI_Auto_Blogger_Pro_Generator {

    private $openai_api_key;
    private $search_api_key;
    private $settings;

    public function __construct() {
        $this->load_settings();
    }

    private function load_settings() {
        $this->settings = get_option('ai_auto_blogger_pro_settings', array());
        $this->openai_api_key = $this->settings['openai_api_key'] ?? '';
        $this->search_api_key = $this->settings['search_api_key'] ?? '';
    }

    /**
     * Generate a new blog post
     *
     * @param string $topic Optional topic override
     * @return int|WP_Error Post ID on success, WP_Error on failure
     */
    public function generate_post($topic = '') {
        
        // 1. Search web for recent content
        $search_results = $this->search_web($topic);
        if (is_wp_error($search_results)) {
            return $search_results;
        }

        // 2. Generate content with AI
        $generated_content = $this->generate_content($topic, $search_results);
        if (is_wp_error($generated_content)) {
            return $generated_content;
        }

        // 3. Create WordPress post
        $post_id = $this->create_post($generated_content);
        if (is_wp_error($post_id)) {
            return $post_id;
        }

        // 4. Log generation
        $this->log_generation($post_id, $topic, $generated_content);

        return $post_id;
    }

    private function search_web($topic) {
        // Implement web search using configured API
        $search_query = $topic ?: $this->get_random_topic();
        
        $response = wp_remote_post('https://api.tavily.com/search', array(
            'headers' => array('Content-Type' => 'application/json'),
            'body'    => json_encode(array(
                'api_key' => $this->search_api_key,
                'query'   => $search_query . ' latest news',
                'max_results' => 5
            ))
        ));

        if (is_wp_error($response)) {
            return $response;
        }

        $body = json_decode(wp_remote_retrieve_body($response), true);
        return $body['results'] ?? array();
    }

    private function generate_content($topic, $search_results) {
        
        $sources_text = $this->format_sources($search_results);
        $prompt = $this->build_prompt($topic, $sources_text);

        $response = wp_remote_post('https://api.openai.com/v1/chat/completions', array(
            'headers' => array(
                'Content-Type' => 'application/json',
                'Authorization' => 'Bearer ' . $this->openai_api_key
            ),
            'body' => json_encode(array(
                'model' => $this->settings['ai_model'] ?? 'gpt-4-turbo-preview',
                'messages' => array(
                    array('role' => 'system', 'content' => 'You are a professional blog writer. Respond only with valid JSON.'),
                    array('role' => 'user', 'content' => $prompt)
                ),
                'temperature' => 0.7,
                'max_tokens' => 3000
            )),
            'timeout' => 60
        ));

        if (is_wp_error($response)) {
            return $response;
        }

        $body = json_decode(wp_remote_retrieve_body($response), true);
        $content_json = $body['choices'][0]['message']['content'] ?? '';
        
        // Parse JSON response
        $content = json_decode($content_json, true);
        if (json_last_error() !== JSON_ERROR_NONE) {
            return new WP_Error('json_parse_error', 'Failed to parse AI response');
        }

        return $content;
    }

    private function create_post($content) {
        
        $post_data = array(
            'post_title'   => wp_strip_all_tags($content['title']),
            'post_content' => wp_kses_post($content['content']),
            'post_excerpt' => wp_strip_all_tags($content['excerpt']),
            'post_status'  => $this->settings['auto_publish'] ? 'publish' : 'draft',
            'post_author'  => $this->settings['post_author'] ?? 1,
            'post_category' => $this->get_category_ids($content['suggested_category'] ?? ''),
            'tags_input'   => $content['tags'] ?? array(),
            'meta_input'   => array(
                '_ai_generated' => 1,
                '_ai_model' => $this->settings['ai_model'] ?? 'gpt-4',
                '_ai_generation_time' => current_time('mysql'),
                '_yoast_wpseo_metadesc' => $content['meta_description'] ?? '',
                '_yoast_wpseo_focuskw' => implode(', ', $content['keywords'] ?? array())
            )
        );

        $post_id = wp_insert_post($post_data);
        
        if (is_wp_error($post_id)) {
            return $post_id;
        }

        return $post_id;
    }

    private function build_prompt($topic, $sources_text) {
        $language = $this->settings['language'] ?? 'English';
        $style = $this->settings['content_style'] ?? 'professional';
        $min_words = $this->settings['min_word_count'] ?? 500;
        $max_words = $this->settings['max_word_count'] ?? 1500;

        return "Write a comprehensive blog post in {$language} about: {$topic}

Based on these recent sources:
{$sources_text}

Requirements:
- {$style} tone
- {$min_words}-{$max_words} words
- HTML formatting with <h2>, <h3>, <p>, <ul>, <ol>
- Practical insights and examples
- SEO-friendly

Output as JSON:
{
  \"title\": \"...\",
  \"content\": \"<h2>...</h2><p>...</p>\",
  \"excerpt\": \"...\",
  \"meta_description\": \"...\",
  \"keywords\": [...],
  \"tags\": [...],
  \"suggested_category\": \"...\"
}";
    }

    private function get_random_topic() {
        $topics = explode(',', $this->settings['topics'] ?? 'technology,business,lifestyle');
        return trim($topics[array_rand($topics)]);
    }

    private function format_sources($results) {
        $formatted = array();
        foreach ($results as $result) {
            $formatted[] = "Title: {$result['title']}\nSnippet: {$result['content']}\nURL: {$result['url']}";
        }
        return implode("\n\n", $formatted);
    }

    private function get_category_ids($category_slug) {
        if (empty($category_slug)) {
            return array($this->settings['default_category'] ?? 1);
        }
        
        $category = get_category_by_slug($category_slug);
        return $category ? array($category->term_id) : array($this->settings['default_category'] ?? 1);
    }

    private function log_generation($post_id, $topic, $content) {
        global $wpdb;
        
        $wpdb->insert(
            $wpdb->prefix . 'ai_generation_logs',
            array(
                'post_id' => $post_id,
                'search_query' => $topic,
                'ai_model' => $this->settings['ai_model'] ?? 'gpt-4',
                'status' => 'success',
                'tokens_used' => $content['tokens_used'] ?? 0,
                'created_at' => current_time('mysql')
            ),
            array('%d', '%s', '%s', '%s', '%d', '%s')
        );
    }
}
```
